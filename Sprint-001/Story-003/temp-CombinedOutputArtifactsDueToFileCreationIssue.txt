// source/Loan.java
package library.lending;

import java.time.LocalDate;

public class Loan {
    private final String memberId;
    private final String bookId;
    private final LocalDate loanDate;
    private final LocalDate dueDate;
    private boolean isActive;

    public Loan(String memberId, String bookId, LocalDate loanDate, LocalDate dueDate) {
        if (memberId == null || memberId.isEmpty()) {
            throw new IllegalArgumentException("Member ID cannot be null or empty");
        }
        if (bookId == null || bookId.isEmpty()) {
            throw new IllegalArgumentException("Book ID cannot be null or empty");
        }
        if (loanDate == null) {
            throw new IllegalArgumentException("Loan date cannot be null");
        }
        if (dueDate == null) {
            throw new IllegalArgumentException("Due date cannot be null");
        }
        if (loanDate.isAfter(dueDate)) {
            throw new IllegalArgumentException("Loan date cannot be after due date");
        }

        this.memberId = memberId;
        this.bookId = bookId;
        this.loanDate = loanDate;
        this.dueDate = dueDate;
        this.isActive = true;
    }

    public String getMemberId() {
        return memberId;
    }

    public String getBookId() {
        return bookId;
    }

    public LocalDate getLoanDate() {
        return loanDate;
    }

    public LocalDate getDueDate() {
        return due
    }

    public boolean isActive() {
        return isActive;
    }

    public void close() {
        if (!isActive) {
            throw new IllegalStateException("Loan cannot be closed if it is not active");
        }
        isActive = false;
    }
}


// tests/LoanTest.java
package library.lending;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class LoanTest {

    @Test
    public void testValidLoan() {
        Loan loan = new Loan("M123", "B456", LocalDate.now(), LocalDate.now().plusDays(14));
        assertEquals("M123", loan.getMemberId());
        assertEquals("B456", loan.getBookId());
        assertTrue(loan.isActive());
    }

    @Test
    public void testInvalidMemberId() {
        assertThrows(IllegalArgumentException.class, () -> new Loan("", "B456", LocalDate.now(), LocalDate.now().plusDays(14)));
    }

    @Test
    public void testInvalidBookId() {
        assertThrows(IllegalArgumentException.class, () -> new Loan("M123", "", LocalDate.now(), LocalDate.now().plusDays(14)));
    }

    @Test
    public void testInvalidLoanDate() {
        assertThrows(IllegalArgumentException.class, () -> new Loan("M123", "B456", null, LocalDate.now().plusDays(14)));
    }

    @Test
    public void testInvalidDueDate() {
        assertThrows(IllegalArgumentException.class, () -> new Loan("M123", "B456", LocalDate.now(), null));
    }

    @Test
    public void testLoanDateAfterDueDate() {
        assertThrows(IllegalArgumentException.class, () -> new Loan("M123", "B456", LocalDate.now().plusDays(14), LocalDate.now()));
    }

    @Test
    public void testCloseLoan() {
        Loan loan = new Loan("M123", "B456", LocalDate.now(), LocalDate.now().plusDays(14));
        loan.close();
        assertFalse(loan.isActive());
    }

    @Test
    public void testCloseInactiveLoan() {
        Loan loan = new Loan("M123", "B456", LocalDate.now(), LocalDate.now().plusDays(14));
        loan.close();
        assertThrows(IllegalStateException.class, () -> loan.close());
    }
}



# README.md

## Loan Aggregate Implementation

The Loan Aggregate represents the lending of one Book to one Member.

### Key Features

- Associates exactly one Member and one Book
- Stores Loan Date and Due Date
- Tracks Loan Status (Active/Inactive)
- Enforces business rules for valid loans

### Implementation Details

- The Loan class enforces all business rules through its constructor
- Loan status is managed through the `close()` method
- All operations preserve the integrity of the Loan Aggregate

### Testing

- Comprehensive unit tests cover all validation scenarios
- Tests verify proper behavior for valid and invalid loan requests
- Tests ensure correct handling of loan closure
